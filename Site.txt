function generateSite() {
  const business = document.getElementById("businessType").value;
  const desc = document.getElementById("description").value;
  const theme = document.getElementById("theme").value;
  const result = document.getElementById("result");

  if (!business) return alert("Ajoute un type d'entreprise !");

  const themes = {
    luxury: "bg-gradient-to-br from-yellow-500 to-yellow-700 text-white",
    minimal: "bg-white text-gray-900 border border-gray-300",
    dark: "bg-gradient-to-br from-gray-900 to-gray-700 text-white",
    ocean: "bg-gradient-to-br from-cyan-500 to-blue-700 text-white",
  };

  const sections = [
    {
      title: "Présentation",
      text: desc || `Découvrez ${business}, un service haut de gamme conçu pour satisfaire vos besoins.`
    },
    {
      title: "Pourquoi nous choisir ?",
      text: `${business} offre une expertise inégalée, des solutions personnalisées et une satisfaction garantie.`
    },
    {
      title: "Services",
      text: `Nos services sont adaptés à vos exigences et garantissent une expérience de qualité.`
    },
    {
      title: "Avis Clients",
      text: `"Un service incroyable, je recommande ${business} !" - Client satisfait`
    },
    {
      title: "Contact",
      text: `Contactez l'équipe ${business} pour démarrer votre projet.`
    }
  ];

  let html = `<div class='p-10 rounded-3xl shadow-xl ${themes[theme]} fade-in'>`;
  html += `<h2 class='text-4xl font-extrabold mb-6'>Site Pro — ${business}</h2>`;

  sections.forEach(section => {
    html += `<div class='mb-8'>
      <h3 class='text-2xl font-bold mb-2'>${section.title}</h3>
      <p class='text-lg'>${section.text}</p>
    </div>`;
  });

  html += `<button onclick='exportHTML()' class='mt-6 p-4 rounded-xl bg-black bg-opacity-30 backdrop-blur-md'>Exporter en HTML</button>`;

  html += `</div>`;

  result.innerHTML = html;
}

function exportHTML() {
  const content = document.getElementById("result").innerHTML;
  const file = new Blob([content], { type: "text/html" });
  const a = document.createElement("a");
  a.href = URL.createObjectURL(file);
  a.download = "site_genere.html";
  a.click();
}
